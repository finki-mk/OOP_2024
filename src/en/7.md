### 1.1. Hotel Reservation

Define a class `HotelReservation` for a hotel reservation. The
reservation stores the number of days, number of people, and the first
and last name of the contact person. Assume that the price per person
per day is 25 euros.

Define a function `price()` that returns the total reservation price.
Also define `price(int amount)` that returns the change if the user pays
a given amount.

Derive a class `HalfBoardHotelReservation` (room + breakfast). Breakfast
costs 5 euros per person per day. Appropriately override the `price(int amount)` function.

Define a class `Hotel` with name and balance. Implement:

int pay(HotelReservation &hr, int amount);

With this function, a payment should be made for a given hotel reservation. If the payment exceeds the requested amount, the function should return how much money should be returned to the user making the payment. The payment should be added to the hotel's balance.

What would happen if the argument is NOT passed by reference?

``` cpp
#include<iostream>
#include<cstring>

using namespace std;

class HotelReservation {
protected:
    int days; // number of days
    int people; // number of people
    string name; // first name
    string surname; // last name

public:
    HotelReservation(string name, string surname, int days, int people) {
        this->name = name;
        this->surname = surname;
        this->days = days;
        this->people = people;
    }

    virtual int price() {
        return days * people * 25;
    }

    virtual int price(int amount) {
        if (amount >= price())
            return amount - price();
        else {
            cout << "You need to pay " << price() << endl;
            return -1;
        }
    }
};

class HalfBoardHotelReservation : public HotelReservation {
public:
    HalfBoardHotelReservation(string name, string surname, int days, int people)
        : HotelReservation(name, surname, days, people) {
    }

    // overriding price(int amount)
    int price(int amount) {
        int totalPrice = HotelReservation::price() + people * 5;
        if (amount >= totalPrice)
            return amount - totalPrice;
        else {
            cout << "You need to pay " << totalPrice << endl;
            return -1;
        }
    }
};

class Hotel {
private:
    string name;
    int balance;

public:
    Hotel(string name) {
        this->name = name;
        balance = 0;
    }

    // reference to base class allows polymorphism
    int pay(HotelReservation &hr, int uplata) {
        int change = hr.price(uplata); // polymorphic call
        if (change != -1)
            balance += uplata - change;
        return change;
    }
};

int main() {
    Hotel h("Bristol");
    HotelReservation *hr1 = new HotelReservation("Petko", "Petkovski", 5, 5);
    int cena = h.pay(*hr1, 1000);
    if (cena != -1)
        cout << "Change : " << cena << endl;

    HalfBoardHotelReservation *hr2 =
            new HalfBoardHotelReservation("Risto", "Ristovski", 5, 5);
    cena = h.pay(*hr2, 1000);
    if (cena != -1)
        cout << "Change : " << cena << endl;

    HotelReservation *hr3 = new HalfBoardHotelReservation("Ana", "Anovska", 4, 2);
    cena = h.pay(*hr3, 100);
    if (cena != -1)
        cout << "Change : " << cena << endl;

    HalfBoardHotelReservation hr4("Tome", "Tomovski", 5, 3);
    cena = h.pay(hr4, 1000);
    if (cena != -1)
        cout << "Change : " << cena << endl;
}
```

### 1.2. Social Network Groups

A class `User` , which stores information about a username, the name of the company they work for, and the level of their position in the company (a number from 1 to 10). All necessary constructors, operators, set and get methods, as well as print and read operators, are defined in the class. This class should not be changed.

Define class `Group`that will represent a group of users. It stores information about:
 - Group members (dynamically allocated array of objects of the User class)
 - Group size (number of elements in the array)
 - Group name (string)

Implement the following for the class:
 - The necessary constructors and destructor
 - method void addMember (User & u) - which will add a new member to the group. Do not allow adding a new member with an existing username.
 - method double rating() - which will calculate the rating of the group using the formula: (10 - average_level_of_the_members_positions) * number_of_members / 100.

Implement a class `PrivateGroup`in which, in addition to the basic information about a group, the following will also be stored:
 - Capacity of the private group (integer). The capacity is the same for all private groups and initially has a value of 10. It can be changed.

Implement the necessary constructors and destructor for the class.

In the `PrivateGroup` class , override the methods from the `Group` class in the following way:
 - method `void addMember (User & u)` which will add a new member to the group. Do not allow exceeding the capacity.
 - method `double rating()` which will calculate the rating of the group using the formula: `(10 - average_level_of_the_members_positions) * (number_of_members / group_capacity) * private_group_coefficient`. The private group coefficient is the same for all private groups and is 0.8. It cannot be changed!

Define a print operator << for both classes. Check the print format for both groups in the first two test cases.

``` cpp
#include<cstring>
#include<iostream>

using namespace std;

class User {
    string username;
    string companyName;
    int position;

    void copy(const User &u) {
        this->username = u.username;
        this->companyName = u.companyName;
        this->position = u.position;
    }

public:
    User(const string username = "", const string companyName = "", int position = 0) {
        this->username = username;
        this->companyName = companyName;
        this->position = position;
    }

    User(const User &u) {
        copy(u);
    }

    User &operator=(const User &u) {
        if (this != &u) {
            copy(u);
        }
        return *this;
    }

    ~User() {
    }

    string getUsername() const {
        return username;
    }

    void setUsername(string username) {
        this->username = username;
    }

    string getCompanyName() const {
        return companyName;
    }

    void setCompanyName(string companyName) {
        this->companyName = companyName;
    }

    int getPosition() const {
        return position;
    }

    void setPosition(int position) {
        this->position = position;
    }


    friend ostream &operator<<(ostream &os, const User &user) {
        return os << "Username: " << user.username
                  << " Company name: " << user.companyName
                  << " Position: " << user.position;
    }

    friend istream &operator>>(istream &in, User &user) {
        return in >> user.username >> user.companyName >> user.position;
    }

    bool operator==(User &u) {
        return this->username==u.username;
    }
};

class Group {
protected:
    User *users;
    int n;
    string name;

    void copy(const Group &g) {
        this->n = g.n;
        this->name = g.name;
        users = new User[n];
        for (int i = 0; i < n; i++)
            users[i] = g.users[i];
    }

public:
    Group(string name = "") {
        this->name = name;
        users = new User[0];
        n = 0;
    }

    Group(const Group &g) {
        copy(g);
    }

    Group &operator=(const Group &g) {
        if (this != &g) {
            delete[] users;
            copy(g);
        }
        return *this;
    }

    ~Group() {
        delete[] users;
    }

    virtual void addMember(User &u) {
        for (int i = 0; i < n; i++) {
            if (users[i] == u)
                return;
        }
        User *tmp = new User[n + 1];
        for (int i = 0; i < n; i++)
            tmp[i] = users[i];
        tmp[n++] = u;
        delete[] users;
        users = tmp;
    }

    double getAverageLevel() const {
        if (n == 0)
            return 0;
        int sum = 0;
        for (int i = 0; i < n; i++)
            sum += users[i].getPosition();
        return double(sum) / n;
    }

    virtual double rating() const {
        return (10 - getAverageLevel()) * n / 100.0;
    }

    friend ostream &operator<<(ostream &out, const Group &g) {
        out << "Group: " << g.name << endl;
        out << "Members: " << g.n << endl;
        out << "Rating: " << g.rating() << endl;
        out << "Members list: " << endl;
        if (g.n == 0) {
            out << "EMPTY" << endl;
        } else {
            for (int i = 0; i < g.n; i++) {
                out << i + 1 << ". " << g.users[i] << endl;
            }
        }

        return out;
    }

};

class PrivateGroup : public Group {
public:
    static int CAPACITY;
    static const float COEFFICIENT;

    PrivateGroup(string name) : Group(name) {};

    PrivateGroup(const PrivateGroup &pg) : Group(pg) {}

    PrivateGroup &operator=(const PrivateGroup &pr) {
        if (this != &pr) {
            Group::operator=(pr);
        }
        return *this;
    }

    void addMember(User &u) {
        if (this->n == CAPACITY) {
            return;
        }
        Group::addMember(u);
    }

    double rating() const {
        return (10 - getAverageLevel()) * (double(n) / CAPACITY) * COEFFICIENT;
    }

    static void setCapacity(int capacity) {
        CAPACITY = capacity;
    }

    static int getCapacity() {
        return CAPACITY;
    }

    friend ostream &operator<<(ostream &out, const PrivateGroup &g) {
        out << "Private ";
        Group group(g);
        out << group;
        return out;
    }

};

int PrivateGroup::CAPACITY = 10;
const float PrivateGroup::COEFFICIENT = 0.8f;

int main() {

    cout << "INTEGRATION TEST" << endl;
    char name[50];
    int nGroups;
    cin >> nGroups;
    Group **groups = new Group *[nGroups];
    for (int i = 0; i < nGroups; i++) {
        int type;
        cin >> type;
        cin >> name;
        if (type == 1) { // normal group
            groups[i] = new Group(name);
        } else { // private group
            groups[i] = new PrivateGroup(name);
        }
        int nUsers;
        cin >> nUsers;
        for (int j = 0; j < nUsers; j++) {
            User u;
            cin >> u;

            groups[i]->addMember(u);

        }
    }

    cout << "BEFORE CHANGE OF PRIVATE GROUP COEFFICIENT" << endl;
    for (int i = 0; i < nGroups; ++i) {
        cout << *groups[i];
    }
    PrivateGroup::setCapacity(15);
    cout << "AFTER CHANGE OF PRIVATE GROUP COEFFICIENT" << endl;
    for (int i = 0; i < nGroups; ++i) {
        cout << *groups[i];
    }

}
```
