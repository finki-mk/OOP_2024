### 1.1. Hotel Reservation

Define a class `HotelReservation` for a hotel reservation. The
reservation stores the number of days, number of people, and the first
and last name of the contact person. Assume that the price per person
per day is 25 euros.

Define a function `price()` that returns the total reservation price.
Also define `price(int amount)` that returns the change if the user pays
a given amount.

Derive a class `HalfBoardHotelReservation` (room + breakfast). Breakfast
costs 5 euros per person per day. Override `price(int amount)`.

Define a class `Hotel` with name and balance. Implement:

int pay(HotelReservation &hr, int amount);

This processes payment and returns change if overpaid. Payment is added
to the balance.

What would happen if the argument is NOT passed by reference?

``` cpp
#include<iostream>
#include<cstring>

using namespace std;

class HotelReservation {
protected:
    int denovi; // number of days
    int broj_lica; // number of people
    string ime; // first name
    string prezime; // last name

public:
    HotelReservation(string ime, string prezime, int denovi, int broj_lica) {
        this->ime = ime;
        this->prezime = prezime;
        this->denovi = denovi;
        this->broj_lica = broj_lica;
    }

    virtual int price() {
        return denovi * broj_lica * 25;
    }

    virtual int price(int amount) {
        if (amount >= price())
            return amount - price();
        else {
            cout << "You need to pay " << price() << endl;
            return -1;
        }
    }
};

class HalfBoardHotelReservation : public HotelReservation {
public:
    HalfBoardHotelReservation(string ime, string prezime, int denovi, int broj_lica)
        : HotelReservation(ime, prezime, denovi, broj_lica) {}

    // overriding price(int amount)
    int price(int amount) {
        int cena = HotelReservation::price() + broj_lica * 5;
        if (amount >= cena)
            return amount - cena;
        else {
            cout << "You need to pay " << cena << endl;
            return -1;
        }
    }
};

class Hotel {
private:
    string ime;
    int saldo;
public:
    Hotel(string ime) {
        this->ime = ime;
        saldo = 0;
    }

    // reference to base class allows polymorphism
    int pay(HotelReservation &hr, int uplata) {
        int kusur = hr.price(uplata); // polymorphic call
        if (kusur != -1)
            saldo += uplata - kusur;
        return kusur;
    }
};

int main() {

    Hotel h("Bristol");
    HotelReservation *hr1 = new HotelReservation("Petko", "Petkovski", 5, 5);
    int cena = h.pay(*hr1, 1000);
    if (cena != -1)
        cout << "Change : " << cena << endl;

    HalfBoardHotelReservation *hr2 =
            new HalfBoardHotelReservation("Risto", "Ristovski", 5, 5);
    cena = h.pay(*hr2, 1000);
    if (cena != -1)
        cout << "Change : " << cena << endl;

    HotelReservation *hr3 = new HalfBoardHotelReservation("Ana", "Anovska", 4, 2);
    cena = h.pay(*hr3, 100);
    if (cena != -1)
        cout << "Change : " << cena << endl;

    HalfBoardHotelReservation hr4("Tome", "Tomovski", 5, 3);
    cena = h.pay(hr4, 1000);
    if (cena != -1)
        cout << "Change : " << cena << endl;
}
```

### 1.2. Social Network Groups

A class `User` is given (DO NOT modify it).

Define class `Group`: - dynamic array of users - size - name

Implement: - constructors/destructor - addMember (no duplicates) -
rating()

Formula: (10 - average_position) \* number_of_members / 100

Define `PrivateGroup`: - capacity (default 10, changeable)

Override: - addMember (respect capacity) - rating: (10 - avg_position) \* (members / capacity) \* 0.8

``` cpp
// All comments translated

class Group {
protected:
    User *users;
    int n;
    string name;

    void copy(const Group &g) {
        this->n = g.n;
        this->name = g.name;
        users = new User[n];
        for (int i = 0; i < n; i++)
            users[i] = g.users[i];
    }

public:
    Group(string name = "") {
        this->name = name;
        users = new User[0];
        n = 0;
    }

    virtual void addMember(User &u) {
        for (int i = 0; i < n; i++) {
            if (users[i] == u)
                return;
        }
        User *tmp = new User[n + 1];
        for (int i = 0; i < n; i++)
            tmp[i] = users[i];
        tmp[n++] = u;
        delete[] users;
        users = tmp;
    }

    double getAverageLevel() const {
        if (n == 0)
            return 0;
        int sum = 0;
        for (int i = 0; i < n; i++)
            sum += users[i].getPosition();
        return double(sum) / n;
    }

    virtual double rating() const {
        return (10 - getAverageLevel()) * n / 100.0;
    }
};

class PrivateGroup : public Group {
public:
    static int CAPACITY;
    static const float COEFFICIENT;

    PrivateGroup(string name) : Group(name) {};

    void addMember(User &u) {
        if (this->n == CAPACITY)
            return;
        Group::addMember(u);
    }

    double rating() const {
        return (10 - getAverageLevel()) * (double(n) / CAPACITY) * COEFFICIENT;
    }
};

int PrivateGroup::CAPACITY = 10;
const float PrivateGroup::COEFFICIENT = 0.8f;
```
